Data frames

df <- data.frame(
  x = c(3, 4, 5, 6),
  y = c(7, 8, 9, 10),
  z = c(11, 12, 13, 14))
df
##   x  y  z
## 1 3  7 11
## 2 4  8 12
## 3 5  9 13
## 4 6 10 14

How do we take the mean of each column?

# Don't do this:
for (i in seq_along(df)) print(mean(df[,i]))
## [1] 4.5
## [1] 8.5
## [1] 12.5
# Do this:
lapply(df, mean)
## $x
## [1] 4.5
## 
## $y
## [1] 8.5
## 
## $z
## [1] 12.5
# Or this:
purrr::map(df, mean)
## $x
## [1] 4.5
## 
## $y
## [1] 8.5
## 
## $z
## [1] 12.5
